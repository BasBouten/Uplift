        const screens = ['dashboard-screen', 'job-detail-screen', 'troubleshooting-screen', 'documentation-screen'];

        const navItems = {

            'dashboard-screen': document.getElementById('nav-dashboard'),

            'diagnostics-screen': document.getElementById('nav-diagnostics'), // Placeholder for future screen

            'schematics-screen': document.getElementById('nav-schematics'), // Placeholder for future screen

        let currentTroubleshootingStep = 0;

        const totalTroubleshootingSteps = 5;


        function showScreen(screenId) {

            screens.forEach(id => {

                const screen = document.getElementById(id);

                if (screen) {

                    screen.classList.remove('active');

                }

            });

            const activeScreen = document.getElementById(screenId);

            if (activeScreen) {

                activeScreen.classList.add('active');

            }


            // Update active nav item

            for (const navId in navItems) {

                if (navItems[navId]) {

                    navItems[navId].classList.remove('active');

                }

            }

            if (navItems[screenId]) { // Only activate if it's a direct nav screen

                navItems[screenId].classList.add('active');

            } else if (screenId === 'job-detail-screen' || screenId === 'troubleshooting-screen' || screenId === 'documentation-screen') {

                // For sub-screens, keep the 'Home' nav item active or none if preferred

                navItems['dashboard-screen'].classList.add('active');

            }

        }


        function updateTroubleshootingProgress() {

            const progressBar = document.getElementById('troubleshooting-progress');

            const nextStepButton = document.getElementById('next-step-button');

            const steps = document.querySelectorAll('#ar-steps .ar-step');


            if (currentTroubleshootingStep < totalTroubleshootingSteps) {

                steps[currentTroubleshootingStep].classList.add('completed');

                currentTroubleshootingStep++;

                const progress = (currentTroubleshootingStep / totalTroubleshootingSteps) * 100;

                progressBar.style.width = `${progress}%`;

                nextStepButton.textContent = `Next Step (${currentTroubleshootingStep}/${totalTroubleshootingSteps} Completed)`;

            }


            if (currentTroubleshootingStep === totalTroubleshootingSteps) {

                nextStepButton.textContent = 'Complete Troubleshooting';

                nextStepButton.onclick = () => showScreen('documentation-screen');

            }

        }


        function resetTroubleshooting() {

            currentTroubleshootingStep = 0;

            const progressBar = document.getElementById('troubleshooting-progress');

            const nextStepButton = document.getElementById('next-step-button');

            const steps = document.querySelectorAll('#ar-steps .ar-step');


            progressBar.style.width = '0%';

            nextStepButton.textContent = `Next Step (0/${totalTroubleshootingSteps} Completed)`;

            nextStepButton.onclick = updateTroubleshootingProgress;

            steps.forEach(step => step.classList.remove('completed'));


            // Reset AI insights and summary areas

            document.getElementById('ai-insights-area').classList.add('hidden');

            document.getElementById('ai-insights-content').innerHTML = '';

            document.getElementById('ai-summary-area').classList.add('hidden');

        }


        // Function to call Gemini API for more insights

        async function askAIForInsights() {

            const insightsArea = document.getElementById('ai-insights-area');

            const insightsContent = document.getElementById('ai-insights-content');

            const insightsLoading = document.getElementById('ai-insights-loading');


            insightsArea.classList.remove('hidden');

            insightsContent.innerHTML = '';

            insightsLoading.classList.remove('hidden');


            const jobDetails = `Client: MegaLogistics GmbH, Equipment: AS/RS-Conveyor-Robot Interop Failure, Status: Intermittent Communication Loss (AS/RS <-> Conveyor)`;

            const aiDiagnosis = `AI Suggestion: Communication Protocol Mismatch between AS/RS Controller (v2.1) and Conveyor PLC (v4.0.3) detected in 'End-of-Cycle' signal interpretation. Likelihood: High (92%). Recommended Action: Review signal mapping & apply compatibility patch.`;


            const prompt = `As an expert in intralogistics system interoperability, provide more detailed troubleshooting steps and potential root causes for the following scenario, building on the initial AI diagnosis. Focus on common interop issues beyond basic firmware mismatches, such as network latency, data format discrepancies, or timing synchronization problems.

            Job Details: ${jobDetails}

            Initial AI Diagnosis: ${aiDiagnosis}

            Please provide actionable steps and common pitfalls.`;


            let chatHistory = [];

            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };

            const apiKey = ""; // Canvas will provide this at runtime

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


            try {

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const result = await response.json();

                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    insightsContent.innerHTML = text.replace(/\n/g, '<br>'); // Display text with line breaks

                } else {

                    insightsContent.textContent = "Error: Could not get AI insights. Please try again.";

                }

            } catch (error) {

                console.error("Error calling Gemini API:", error);

                insightsContent.textContent = "Error: Failed to connect to AI service. Please check your network.";

            } finally {

                insightsLoading.classList.add('hidden');

            }

        }
